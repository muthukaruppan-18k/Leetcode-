/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isBalanced(TreeNode root) {
        return getHeight(root) != -1;
    }

    private int getHeight(TreeNode node) {
        if (node == null) return 0;

        // 1. Get left height
        int leftHeight = getHeight(node.left);
        if (leftHeight == -1) return -1; // Already unbalanced below

        // 2. Get right height
        int rightHeight = getHeight(node.right);
        if (rightHeight == -1) return -1; // Already unbalanced below

        // 3. Check current node balance
        if (Math.abs(leftHeight - rightHeight) > 1) {
            return -1; // Unbalanced here!
        }

        // 4. Return actual height if balanced (same as maxDepth)
        return Math.max(leftHeight, rightHeight) + 1;
    }
}

int removeDuplicates(int* nums, int numsSize) {
    if (numsSize == 0) return 0;

    int l = 1; // First element is always unique
    for (int i= 1; i< numsSize; i++) {
        if (nums[i] != nums[l - 1]) {
            nums[l] = nums[i];
            l++;
        }
    }
    return l; // Number of unique elements
}
